\3. 手写发布订阅模式

let event = {
  list: {},
  on(key, fn) {
    if(!this.list[key]) {
      this.list[key] = []
    }
    this.list[key].push(fn)
  },
  emit() {
    let key = [].shift.call(arguments)
    fns = this.list[key]
    if(!fns || fns.length === 0) {
      return console.log('该key没有对应的事件')
    }
    fns.forEach(fn => {
      fn.apply(this, arguments)
    })
  }
}
